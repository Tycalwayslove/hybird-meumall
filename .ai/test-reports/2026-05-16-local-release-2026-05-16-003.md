## 命令

```bash
cd /Users/mac/company_code/server-meumall
. .venv/bin/activate && python -m uvicorn app.main:app --host 127.0.0.1 --port 4100

cd /Users/mac/company_code/admin-meumall
pnpm dev --host 127.0.0.1 --port 5173

cd /Users/mac/company_code/hybird-meumall
PORT=3112 HOSTNAME=127.0.0.1 NODE_ENV=production H5_BASE_PATH=/hybird H5_RELEASE_VARIANT=green H5_RELEASE_LABEL='NEW 2026.05.16-003' H5_VERSION='2026.05.16-003' node .next/standalone/server.js

pnpm run ai:release-prepare --version 2026.05.16-003 --channel stable --rollback-version 2026.05.16-green --rollout-percentage 0 --routes /,/category,/cart,/profile --service-base-url http://127.0.0.1:3112 --base-path /hybird
pnpm run ai:prepare-ssr-release --version 2026.05.16-003 --environment prod --service-base-url http://127.0.0.1:3112 --base-path /hybird
pnpm run ai:smoke-ssr-release --plan archives/releases/2026.05.16-003/ssr-release-plan.json
pnpm run ai:register-release --version 2026.05.16-003 --environment prod --service-base-url http://127.0.0.1:3112 --base-path /hybird --rollback-version 2026.05.16-green --rollout-percentage 0 --routes /,/category,/cart,/profile --server-url http://127.0.0.1:4100 --execute
```
